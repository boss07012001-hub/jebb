function handleDrop(ev){

function handleFile(file){

  if(!file) return;

  const ext=file.name.split('.').pop().toLowerCase();

  const reader=new FileReader();

  reader.onload=function(e){

    try {

      let rows=[];

      if(ext==='csv'){

        const text=new TextDecoder('utf-8').decode(e.target.result);

        rows=parseCSV(text);

      } else {

        const wb=XLSX.read(e.target.result,{type:'array'});

        const ws=wb.Sheets[wb.SheetNames[0]];

        rows=XLSX.utils.sheet_to_json(ws,{defval:''});

      }

      processImportRows(rows);

    } catch(err){

      showImportResult('err','檔案解析失敗：'+err.message);

    }

  };

  if(ext==='csv') reader.readAsArrayBuffer(file);

  else reader.readAsArrayBuffer(file);

}

function parseCSV(text){

  const lines=text.trim().split(/\r?\n/);

  if(lines.length<2) return [];

  const headers=lines[0].split(',').map(h=>h.trim().replace(/^"|"$/g,''));

  return lines.slice(1).map(line=>{

    const vals=line.split(',').map(v=>v.trim().replace(/^"|"$/g,''));

    const obj={};

    headers.forEach((h,i)=>{ obj[h]=vals[i]||''; });

    return obj;

  });

}

function processImportRows(rows){

  const area=document.getElementById('import-result-area');

  if(!rows||!rows.length){ showImportResult('err','檔案內無資料行'); return; }

 

  // Flexible column name matching

  const colMap={};

  const sample=rows[0];

  const keys=Object.keys(sample);

  const find=(candidates)=>keys.find(k=>candidates.some(c=>k.includes(c)))||'';

  colMap.date  = find(['日期','date','Date']);

  colMap.type  = find(['類型','type','Type','收支']);

  colMap.cat   = find(['科目','category','Category','分類']);

  colMap.desc  = find(['摘要','description','Description','備註','說明']);

  colMap.amt   = find(['金額','amount','Amount','價格']);

 

  const missing=Object.entries(colMap).filter(([,v])=>!v).map(([k])=>({date:'日期',type:'類型',cat:'科目',desc:'摘要',amt:'金額'}[k]));

  if(missing.length){

    showImportResult('err',`找不到必要欄位：${missing.join('、')}。請確認第一列標題列正確。`);

    return;

  }

 

  const parsed=[];

  const errors=[];

  rows.forEach((row,i)=>{

    const dateRaw=String(row[colMap.date]||'').trim();

    const typeRaw=String(row[colMap.type]||'').trim();

    const catRaw =String(row[colMap.cat]||'').trim();

    const descRaw=String(row[colMap.desc]||'').trim();

    const amtRaw =row[colMap.amt];

 

    // Date

    let date=dateRaw;

    if(!date){ errors.push(`第 ${i+2} 列：日期為空`); return; }

    // Handle Excel serial date numbers

    if(!isNaN(Number(date))){

      const d=XLSX.SSF.parse_date_code(Number(date));

      if(d) date=`${d.y}-${String(d.m).padStart(2,'0')}-${String(d.d).padStart(2,'0')}`;

    }

    if(!/^\d{4}-\d{2}-\d{2}$/.test(date)){ errors.push(`第 ${i+2} 列：日期格式錯誤 (${dateRaw})`); return; }

 

    // Type

    const type = (typeRaw==='收入'||typeRaw.toLowerCase()==='income') ? 'income'

               : (typeRaw==='支出'||typeRaw.toLowerCase()==='expense') ? 'expense'

               : null;

    if(!type){ errors.push(`第 ${i+2} 列：類型應為「收入」或「支出」，得到「${typeRaw}」`); return; }

 

    // Category – try label match, fallback to custom

    const matchedCat=ALL_CAT_LABELS[catRaw];

    const catId  = matchedCat ? matchedCat.id : 'other_'+(type==='income'?'i':'e');

    const catLabel= matchedCat ? matchedCat.label : (catRaw||'其他');

 

    // Amount

    const amt=parseFloat(String(amtRaw).replace(/,/g,''));

    if(isNaN(amt)||amt<=0){ errors.push(`第 ${i+2} 列：金額無效 (${amtRaw})`); return; }

 

    if(!descRaw){ errors.push(`第 ${i+2} 列：摘要為空`); return; }

 

    parsed.push({id:Date.now()+i, date, desc:descRaw, type, catId, catLabel, amount:amt});

  });

 

  pendingImport=parsed;

 

  // Show preview

  const previewRows=parsed.slice(0,8);

  const more=parsed.length>8 ? `<tr><td colspan="5" style="text-align:center;color:var(--text-light);font-size:12px">…還有 ${parsed.length-8} 筆</td></tr>` : '';

  const previewHtml=`

    <div class="import-preview">

      <table>

        <thead><tr><th>日期</th><th>類型</th><th>科目</th><th>摘要</th><th style="text-align:right">金額</th></tr></thead>

        <tbody>

          ${previewRows.map(r=>`<tr>

            <td style="font-family:var(--mono)">${r.date}</td>

            <td><span class="badge-type ${r.type}">${r.type==='income'?'收入':'支出'}</span></td>

            <td>${r.catLabel}</td><td>${r.desc}</td>

            <td style="text-align:right;font-family:var(--mono)">${fmt(r.amount)}</td>

          </tr>`).join('')}

          ${more}

        </tbody>

      </table>

    </div>

    ${errors.length ? `<div class="import-result warn">⚠️ 已跳過 ${errors.length} 列有問題的資料：${errors.slice(0,3).join('；')}${errors.length>3?'…':''}</div>` : ''}

    ${parsed.length>0 ? `

    <div class="import-result ok">✓ 成功解析 <strong>${parsed.length}</strong> 筆記錄，確認後點擊「匯入」</div>

    <div class="import-actions">

      <button class="btn-cancel-imp" onclick="cancelImport()">取消</button>

      <button class="btn-confirm" onclick="confirmImport()">✓ 匯入 ${parsed.length} 筆記錄</button>

    </div>` : `<div class="import-result err">❌ 無可匯入的有效資料</div>`}

  `;

  area.innerHTML=previewHtml;

}

function group(type){

  });

  return m;

}

function fmtR(n, inc){

  // Format a ratio value as a percentage pill; n = numerator, inc = denominator (revenue)

  if(!inc || inc===0) return '<span class="ratio-pill neu">—</span>';

  const pct = (n / inc * 100);

  const label = pct.toFixed(1) + '%';

  const pillCls = pct > 0 ? 'pos' : pct < 0 ? 'neg' : 'neu';

  return `<span class="ratio-pill ${pillCls}">${label}</span>`;

}
